int ans[10],sign;
 
void solve(int num)
{
  int i,t=num%10;
  if(num<=0) return;
  for(i=0;i<=t;i++)
    ans[i]+=sign;
  num/=10;
  int tt=num;
  while(tt!=0)
  {
    ans[tt%10]+=(t+1)*sign;

}
